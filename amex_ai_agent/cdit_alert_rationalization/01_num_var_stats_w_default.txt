SELECT
  a.trans_dt,
  COUNT(*) AS total_auth_records_cnt,
  COUNT(CASE WHEN b.{var} = 0 THEN a.cas_pkey END) AS {var}_zero_value_record_cnt,
  COUNT(CASE WHEN b.{var} IS NULL THEN a.cas_pkey END) AS {var}_null_value_record_cnt,
  COUNT(CASE WHEN b.{var} = {default_value} THEN a.cas_pkey END) AS {var}_default_value_record_cnt,
  ROUND(AVG(CASE WHEN b.{var} <> {default_value} THEN b.{var} END), 2) AS {var}_avg_excluding_default_and_nulls,
  ROUND(SUM(CASE WHEN b.{var} <> {default_value} THEN b.{var} END), 2) AS {var}_sum_excluding_default_and_nulls,
  MIN(CASE WHEN b.{var} <> {default_value} THEN b.{var} END) AS {var}_min_excluding_default_and_nulls,
  MAX(CASE WHEN b.{var} <> {default_value} THEN b.{var} END) AS {var}_max_excluding_default_and_nulls
FROM `axp-lumi.dw.wwcas_authorization` a
LEFT JOIN `{var_table}` b
  ON a.cas_pkey = b.cas_pkey
WHERE
  a.trans_dt BETWEEN DATE_SUB(DATE '{alert_dt}', INTERVAL 30 DAY)
                  AND DATE_ADD(DATE '{alert_dt}', INTERVAL 5 DAY)
  AND a.arm_proc_ind = '1'
  AND a.country_cd IN ('AR','AT','AU','BE','CA','DE','ES','FI','FR','GB','HK','IT','JP','MX','NL','NO','NZ','SE','SG','TH','TW','US','X0','X2','X3','X4')
GROUP BY 1
ORDER BY 1;
